    Mole~ Config

    Provides config settings for main application to use

    // mole.config.js
export default {
	theme: 'theme/',
	model: ['chars', 'tokens'],
	template: 'templates/',
	output: [
		{ css: { file: 'styles.css' } },
		{ ios: { file: 'styles.h' } },
		{ android: { file: 'styles.xml' } }
	]
}
